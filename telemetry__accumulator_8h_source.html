224 bool SendBuffer(Adapter& adapter, Storage&& buffer, std::size_t messageCount) {
225 // Build packet from extracted buffer
226 TypedPacket<MsgType, Storage> packet;
227 packet.messages = std::move(buffer);
228
229 // Encode to wire format
230 m_wireBuffer.clear();
231 if (!EncodeTypedPacket(packet, m_wireBuffer)) {

233 ++m_metrics.sendFailures;

236
237 // Send (potentially blocking syscall - but we don't hold the lock!)
238 if (!adapter.SendBytes(m_wireBuffer.data(), m_wireBuffer.size())) {
239 std::lock_guard<std::mutex> lock(m_mutex);
240 ++m_metrics.sendFailures;
241 return false;
242 }
243
