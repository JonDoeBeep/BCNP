114 MessageIterator(const uint8_t* ptr, std::size_t count)
115 : m_ptr(ptr), m_count(count) {}

121 MsgType operator*() const {

127 MessageIterator& operator++() {
128 if (m_count > 0) {
129 m_ptr += MsgType::kWireSize;
130 m_count--;

132 if (m_count == 0) {
133 m_ptr = nullptr;
134 }
135 return *this;
136 }

139 MessageIterator operator++(int) {
140 MessageIterator tmp = *this;
141 ++(*this);
142 return tmp;
143 }
144
146 bool operator==(const MessageIterator& other) const {
147 if (m_count == 0 && other.m_count == 0) return true;
148 return m_ptr == other.m_ptr && m_count == other.m_count;
149 }
150

153 return !(*this == other);
154 }

330template<typename MsgType, typename Storage>
331bool EncodeTypedPacket(const TypedPacket<MsgType, Storage>& packet, uint8_t* output,
332 std::size_t capacity, std::size_t& bytesWritten) {
333 bytesWritten = 0;
334 if (packet.messages.size() > kMaxMessagesPerPacket || !output) {
335 return false;
336 }
337
338 const std::size_t payloadSize = kHeaderSizeV3 + packet.messages.size() * MsgType::kWireSize;
339 const std::size_t required = payloadSize + kChecksumSize;
340 if (capacity < required) {
341 return false;
342 }
343
344 // V3 Header
345 output[kHeaderMajorIndex] = packet.header.major;
346 output[kHeaderMinorIndex] = packet.header.minor;
347 output[kHeaderFlagsIndex] = packet.header.flags;
348 detail::StoreU16(static_cast<uint16_t>(MsgType::kTypeId), &output[kHeaderMsgTypeIndex]);
349 detail::StoreU16(static_cast<uint16_t>(packet.messages.size()), &output[kHeaderMsgCountIndex]);
350
351 // Encode messages
352 std::size_t offset = kHeaderSizeV3;
353 for (const auto& msg : packet.messages) {
354 if (!msg.Encode(&output[offset], MsgType::kWireSize)) {
355 return false;
356 }
357 offset += MsgType::kWireSize;
358 }
359
360 // CRC32
361 const uint32_t crc = ComputeCrc32(output, payloadSize);
362 detail::StoreU32(crc, &output[payloadSize]);
363
364 bytesWritten = required;
365 return true;
366}

380template<typename MsgType, typename Storage>
381bool EncodeTypedPacket(const TypedPacket<MsgType, Storage>& packet, std::vector<uint8_t>& output) {
382 if (packet.messages.size() > kMaxMessagesPerPacket) {
383 return false;
384 }
385 const std::size_t required = kHeaderSizeV3 + packet.messages.size() * MsgType::kWireSize + kChecksumSize;
386 output.resize(required);
387 std::size_t bytesWritten = 0;
388 if (!EncodeTypedPacket(packet, output.data(), output.size(), bytesWritten)) {
389 return false;
390 }
391 output.resize(bytesWritten);
392 return true;
393}

405template<typename MsgType>
406crab::Option<TypedPacket<MsgType>> DecodeTypedPacket(const PacketView& view) {
407 if (view.header.messageType != MsgType::kTypeId) {
408 return crab::None;
409 }
410
411 TypedPacket<MsgType> packet;
412 packet.header = view.header;
413 packet.messages.reserve(view.header.messageCount);
414
415 const uint8_t* ptr = view.payload.data();
416 for (std::size_t i = 0; i < view.header.messageCount; ++i) {
417 auto msg = MsgType::Decode(ptr, MsgType::kWireSize);
418 if (!msg) {
419 return crab::None;
420 }
421 packet.messages.push_back(*msg);
422 ptr += MsgType::kWireSize;
423 }
424
425 return crab::Some(std::move(packet));
426}
